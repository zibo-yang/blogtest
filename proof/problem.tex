\section*{Setting and Definitions}

We consider nonlinear PDE of the form
\[
\partial_t u + A(u)\partial_x u + B(u) = 0, \quad \text{on } [0,L],
\]
where $u \in \mathbb{R}^n$, with boundary conditions
\[
\begin{pmatrix} u_+(t,0) \\ u_-(t,L) \end{pmatrix}
= G\begin{pmatrix} u_+(t,L) \\ u_-(t,0) \end{pmatrix}.
\]

A natural quantity arising in the stability analysis is the following function. For $M \in \mathbb{R}^{n \times n}$ and $p \in [1,+\infty]$, define
\[
\rho_p(M) := \inf\bigl\{ \|\Delta M \Delta^{-1}\|_p \;\big|\;
\Delta = \mathrm{diag}(\delta_1,\ldots,\delta_n),\; \delta_i > 0 \;\forall i \bigr\},
\]
where $\|\cdot\|_p$ denotes the matrix $p$-norm (operator norm induced by the vector $\ell^p$ norm).

\section*{Background}

It is known that for any $n \in \mathbb{N}$ and $M \in \mathbb{R}^{n\times n}$:
\[
\rho_1(M) \ge \rho_2(M), \qquad \rho_1(M) = \rho_\infty(M),
\]
and for $n \ge 2$ there exists $M \in \mathbb{R}^{n\times n}$ such that $\rho_1(M) > \rho_2(M)$.
Moreover, if $M$ has only non-negative coefficients then $\rho_k(M) = \rho_1(M)$ for all $k \in [1,+\infty]$.

\section*{The Conjectures}

\begin{conjecture}[Conjecture 1]
For any $M \in \mathbb{R}^{n\times n}$,
\[
\rho_2(M) \le \rho_3(M) \le \cdots \le \rho_\infty(M).
\]
\end{conjecture}

\begin{conjecture}[Conjecture 2]
For any $M \in \mathbb{R}^{n\times n}$,
\[
\rho_2(M) \le \rho_k(M), \quad \forall k \in \mathbb{N} \setminus \{0\}.
\]
\end{conjecture}

\begin{conjecture}[Conjecture 3 --- \textbf{To be proved}]
For any $n > 6$ and any $p > 2$ (with $n, p \in \mathbb{N}$), there exists $M \in \mathbb{R}^{n \times n}$ such that
\[
\rho_p(M) < \rho_{p+1}(M).
\]
\end{conjecture}

\section*{Goal}

Prove Conjecture~3: for every integer $n > 6$ and every integer $p > 2$, exhibit an $n \times n$ matrix $M$ (or a family thereof) for which $\rho_p(M) < \rho_{p+1}(M)$.

\bigskip
\noindent\textbf{Note on the matrix $p$-norm.} Recall that for a matrix $A \in \mathbb{R}^{n\times n}$,
\[
\|A\|_p = \sup_{x \neq 0} \frac{\|Ax\|_p}{\|x\|_p},
\]
with $\|x\|_p = \left(\sum_{i=1}^n |x_i|^p\right)^{1/p}$ for $p < \infty$ and $\|x\|_\infty = \max_i |x_i|$.
Special cases: $\|A\|_1 = \max_j \sum_i |a_{ij}|$ (max column sum), $\|A\|_\infty = \max_i \sum_j |a_{ij}|$ (max row sum), and $\|A\|_2 = \sigma_{\max}(A)$ (largest singular value).
